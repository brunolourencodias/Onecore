
CREATE TABLE REGISTERCLIENT(
RST_ID INT IDENTITY PRIMARY KEY,
USS_ID INT NOT NULL,
RST_NAME VARCHAR(80) NOT NULL,
RST_GENRE CHAR(1) NOT NULL,
RST_DATEBIRTH DATE NOT NULL,
RST_DOCUMENT VARCHAR(11) NOT NULL
)
CREATE TABLE USERS(
USS_ID INT PRIMARY KEY,
USS_NAME VARCHAR(80)NOT NULL,
USS_EMAIL VARCHAR(80)NOT NULL,
USS_PASSWORD VARCHAR(80)NOT NULL,
USS_STATUS CHAR(1)NOT NULL
)


CREATE TABLE COMPANY(
CPY_ID INT IDENTITY PRIMARY KEY,
CPY_NAME VARCHAR(80) NOT NULL,
CPY_DOCUMENT VARCHAR(80) NOT NULL,
CPY_END VARCHAR(80) NOT NULL,
CPY_EMAIL VARCHAR(80) NOT NULL,
CPY_TEL VARCHAR(80) NOT NULL,
CPY_DATECREATE DATE NOT NULL,
USS_ID INT
)

CREATE TABLE VAGANCY(
VAGANCY_ID INT IDENTITY PRIMARY KEY,
VG_TITLE VARCHAR(80) NOT NULL,
CPY_ID INT,
VG_DATE_FINALLY DATE NOT NULL,
VG_OBS VARCHAR(MAX) NOT NULL,
VG_STATUS CHAR(1) NOT NULL,
VG_DATECREATE DATE,
VG_VALUE DECIMAL(18,2)NOT NULL,
USS_ID INT NOT NULL,
VG_EMAIL VARCHAR(80) NOT NULL,
VG_TELEFONE VARCHAR(80) NOT NULL,
VG_NAMECOMPANY VARCHAR(80) NOT NULL
)
